## 1. Overview

The `ContinuumAgent` is a Hard Computing (HC) module designed to estimate and remove the global trend (pseudo-continuum) of a stellar spectrum. This process is vital to produce a **normalized spectrum** where the continuum level is fixed at 1.0. It features a dedicated `rvs_mode` that protects the Ca II triplet regions and a **Universal Point Guard** that invalidates fits performed on insufficient data points after iterative rejection.

## 2. Mathematical Methodology

### 2.1 Iterative Sigma-Clipping with Dynamic Smoothing

1. **Dynamic Stiffness:** The spline smoothing parameter is calculated as $s = N \cdot \sigma^2$ to adapt to varying Signal-to-Noise Ratios (SNR).
2. **Iterative Rejection:** Points outside the $\sigma$ boundaries are masked.
3. **Universal Guard:** After all iterations, the system verifies that the final `continuum_mask` contains at least 10 points. If not, the fit is considered statistically insignificant and marked as `failed`.

### 2.2 RVS Specific Logic (Pseudo-continuum Windows)

In the RVS window (847–874 nm), the agent restricts the fit to safe regions (847.0–849.0 nm, 851.0–853.5 nm, 856.0–860.0 nm, and 869.5–874.0 nm) to avoid contamination from the deep Ca II triplet wings in cool stars (K/M types).

## 3. Logical Flags (Predigestion Logic)

- **`continuum_is_stable`**: `True` if the standard deviation of the normalized flux **within the final continuum mask** is $< 0.05$.
- **`high_snr_continuum`**: `True` if the SNR measured on the continuum regions is $> 20$.
- **`fit_diverged`**: `True` if the fit produced non-physical values or if the **Universal Guard** detected fewer than 10 points in the final mask.

## 4. Implementation Template (Python)


```python
import numpy as np
import logging
from typing import Dict, Any, Tuple
from scipy.interpolate import splrep, splev

logger = logging.getLogger(__name__)

# RVS pseudo-continuum windows (nm) — conservative red edge for cool stars
RVS_CONTINUUM_WINDOWS = [
    (847.0, 849.0), (851.0, 853.5), (856.0, 860.0), (869.5, 874.0)
]

class ContinuumAgent:
    """
    HC Module to normalize stellar spectra using iterative sigma-clipping splines.
    """

    def __init__(
        self,
        sigma_lower: float = 2.0,
        sigma_upper: float = 3.0,
        iterations: int = 5,
        rvs_mode: bool = False
    ):
        self.sigma_lower = sigma_lower
        self.sigma_upper = sigma_upper
        self.iterations  = iterations
        self.rvs_mode    = rvs_mode

    def _build_rvs_window_mask(self, wave: np.ndarray) -> np.ndarray:
        mask = np.zeros(len(wave), dtype=bool)
        for w_min, w_max in RVS_CONTINUUM_WINDOWS:
            mask |= (wave >= w_min) & (wave <= w_max)
        return mask

    def _sigma_clip_fit(
        self,
        wave: np.ndarray,
        flux: np.ndarray,
        initial_mask: np.ndarray = None
    ) -> Tuple[np.ndarray, np.ndarray]:
        mask = initial_mask.copy() if initial_mask is not None else np.ones(len(flux), dtype=bool)
        continuum = np.ones_like(flux)

        for i in range(self.iterations):
            if np.sum(mask) < 10: break

            residuals_current = flux[mask] - np.mean(flux[mask])
            sigma_est = np.std(residuals_current)
            s_factor  = np.sum(mask) * (sigma_est ** 2)

            try:
                tck = splrep(wave[mask], flux[mask], s=s_factor)
                continuum = splev(wave, tck)
            except Exception:
                break

            residuals = flux - continuum
            std = np.std(residuals[mask])
            if std == 0: break

            new_mask = (residuals > -self.sigma_lower * std) & (residuals < self.sigma_upper * std)
            if initial_mask is not None:
                new_mask &= initial_mask

            if np.array_equal(mask, new_mask): break
            mask = new_mask

        return continuum, mask

    def process(self, wavelength: np.ndarray, flux: np.ndarray, source_id: str = "UNKNOWN") -> Dict[str, Any]:
        res = {
            "module": "ContinuumAgent", "status": "failed",
            "measurements": {"normalized_flux": None, "continuum_model": None, "mean_snr": 0.0},
            "logical_flags": {"continuum_is_stable": False, "high_snr_continuum": False, "fit_diverged": False}
        }

        try:
            # 1. Initial Masking (RVS Mode)
            initial_mask = self._build_rvs_window_mask(wavelength) if self.rvs_mode else None
            
            # 2. Fit execution
            continuum_model, continuum_mask = self._sigma_clip_fit(wavelength, flux, initial_mask)

            # 3. Universal Guard: Insufficient points after sigma-clipping
            if np.sum(continuum_mask) < 10:
                logger.error(f"[{source_id}] Final continuum mask has fewer than 10 points. Fit aborted.")
                res["logical_flags"]["fit_diverged"] = True
                return res

            # 4. Guard: Non-physical continuum
            if np.any(continuum_model <= 0):
                logger.error(f"[{source_id}] Continuum model contains non-positive values.")
                res["logical_flags"]["fit_diverged"] = True
                return res

            # 5. Normalization and Metrics
            norm_flux = flux / continuum_model
            snr = float(np.mean(flux[continuum_mask]) / np.std((flux - continuum_model)[continuum_mask]))
            stability_std = float(np.std(norm_flux[continuum_mask]))

            res["measurements"] = {
                "normalized_flux": norm_flux.tolist(),
                "continuum_model": continuum_model.tolist(),
                "mean_snr": round(snr, 2)
            }
            res["logical_flags"] = {
                "continuum_is_stable": bool(stability_std < 0.05),
                "high_snr_continuum": bool(snr > 20),
                "fit_diverged": False
            }
            res["status"] = "success"

        except Exception as e:
            logger.error(f"[{source_id}] ContinuumAgent critical failure: {e}")

        return res
```
